        \item $AB = (2,1,1) $
        \item $AC = (1,-1,2) $
        \item $AD = (0,-2,3) $
    \end{itemize}

    $ Area = \norm*{AB \cdot (AC \crossproduct AD)} = \norm*{det(\begin{vmatrix}
        2 & 1 & 1 \\
        1 & -1 & 2 \\
        0 & -2 & 3
    \end{vmatrix})} = |2\cdot(-3-(-4)) - 3 + (-2) = 2\cdot(7) -3 -2| = 9 $
