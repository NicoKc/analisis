        \item $x^2 + y^2 = 4 \wedge z = x \cdot y$

        \quad $ r(t) = \left\{
            \begin{array}{ll}
                x = 2\cos(t) \\
                y = 2\sin(t) \\

            \end{array}
            \right.$

        \item $ x^2 + y^2 = 1 \wedge y + z = 2$

            \quad $ r(t) = \left\{
                \begin{array}{ll}
                    x = \cos(t) \\
                    y = \sin(t) \\
                    z = 2 - \sin(t) \\
                    0 \leq t \leq 4\pi
                \end{array}
                \right.$

            \includegraphics[scale=0.4]{ej12/resources/1b.png} $ $

        \item $ z = \sqrt{x^2+y^2} \wedge z = 1 + y$

            $ 1 + y = \sqrt{x^2 + y^2} $

            $ (1 + y)^2 = |x^2 + y^2| $

            $ y^2 + 2y + 1 = \abs*{x^2 + y^2} $

            $ y^2 + 2y + 1 = x^2 + y^2 $

            $ y = \frac{x^2 - 1}{2} $

            \quad $ r(t) = \left\{
                \begin{array}{ll}
                    x = t \\
                    y = \frac{t^2 - 1}{2} \\
                    z = \frac{t^2 - 1}{2} + 1 \\
                \end{array}
                \right.$

            \includegraphics[scale=0.4]{ej12/resources/c.png} $ $

    \end{enumerate}
