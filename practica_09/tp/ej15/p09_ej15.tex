            $ 9 - \rho^2\sin^2(\phi)(\cos^(\theta) + \sin^2(\theta)) = $

            $ 9 - \rho^2\sin^2(\phi) $

            $ \int_0^3 (\int_0^{2\pi} (\int_{\frac{\pi}{2}}^{\pi} 9 - x^2 - y^2 d\phi ) d\theta ) d\rho = $

            $ \int_0^3 (\int_0^{2\pi} (\int_{\frac{\pi}{2}}^{\pi} (9 - \rho^2\sin^2(\phi))\rho^2\sin(\phi) d\phi ) d\theta ) d\rho = $
