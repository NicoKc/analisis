    \begin{enumerate}
        \item $f(x,y) = (x^2, e^x)$
        
            $ f(x,y) = (g(x), h(x)) $

            \begin{itemize}
                \item $g(x)=x^2$ es continua
                \item $h(x)=e^x$ es continua
            \end{itemize}

        \item $f(x,y) = (\frac{\sin(x^2+y^2)}{x^2+y^2}, \frac{e^{x^2+y^2}-1}{x^2+y^2})$

            No es continua en el (0,0)

    \end{enumerate}
